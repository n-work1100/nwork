--1--
Зайти в режим глобальної конфігурації
_____________________________________
* enable
* configure terminal
** configure t

--2--
Зайти в режим конфігурації інтерфейсу
_____________________________________
* enable
* configure terminal
* interface vlan 1

--3--
Зайти в режим конфігурації лінії
_____________________________________
* enable
* configure terminal
* line console 0

--4--
Вийти в привелігований режим
_____________________________________
* end / Ctrl + Z

--5--
Перейти з режиму -line в -if
_____________________________________
* interface FastEthernet 0/1
** interface vlan 1

--6--
Перейти з режиму -if в -line
_____________________________________
* line console 1

--7--
Увійти в режим конфігурації лінії VTY
_____________________________________
* configure terminal
* line vty 0 15

--8--
Встановити пароль для користувацького режиму EXEC
_____________________________________
* enable
* configure terminal
* line console 0
* password cisco
* login
* end

--9--
Захист доступу до привилегійованого режиму
_____________________________________
* enable
* configure terminal
* enable secret class
* exit

--10--
Поставити пароль на VTY
_____________________________________
* enable
* configure terminal
* line vty 0 15
* password cisco
* login
* end

--11--
Подивитись ip адреси youtube.com
_____________________________________
* nslookup youtube.com

--12--
Зашифрувати паролі
_____________________________________
* enable
* configure terminal
* service password-encrypting

--13--
Перевірити чи зашифровані паролі
_____________________________________
# show running-config

--14--
Створити банерне повідомлення дня
_____________________________________
* enable
* configure terminal
* banner motd #Authorized Access Only#

--15--
Переглянути файл стартової конфігурації
_____________________________________
show startup-config

--16--
Видалити стартову конфігурацію
_____________________________________
erase startup-config

--17--
Перезавантажити пристрій
_____________________________________
reload

--18--
Зберегти зміни для стартової конфігурації
_____________________________________
copy running-config startup-config

--19--
Включити інтерфейс
_____________________________________
no shutdown

--20--
Показати ip інтерфейсів
_____________________________________
show ip interface brief

--21--
Подивитись маршрути роутера
_____________________________________
* show ip route

--22--
Додати маршрут для роутера
_____________________________________
* ip route 20.20.20.0 255.255.255.0 194.44.1.2

--23--

_____________________________________
tracert 0.0.0.0

--24--
Почати маршрутизацію по ipv6
_____________________________________
* ipv6 unicast-routing

--25--
Задать ipv6 адресу
_____________________________________
* ipv6 route 2001:db8:acad:1::/64 2001:db8:acad:2::2

--26--
Налаштувати маршрут по замовчуванню
_____________________________________
Router(config)# ip route 0.0.0.0 0.0.0.0 {ip-address | exit-intf}
* ip route 0.0.0.0 0.0.0.0 10.10.10.1
** ip route 0.0.0.0 0.0.0.0 GigabitEthernet 0/0

--27--
Налаштувати безпосередньо підключений плаваючий статичний маршрут за замовчуванням IPv4.
Він повинен мати адміністративну відстань 5.
_____________________________________
(config)#ip route 0.0.0.0 0.0.0.0 serial 0/0/1 5

--28--
Налаштувати статичний маршрут за замовчуванням для наступного стрибка.
_____________________________________
(config)# ipv6 route ::/0 2001:db8:a:1::1
